def ordenar_each
    
    @ordenado= []
    aux_self=self.clone
    
    (0..self.size-1).each{|i| min=aux_self.min; @ordenado << min; aux_self.delete(min)}
    
    @ordenado
end

def ordenar_for
    
    @vector=self
    
    for i in (0..@vector.size-1)
        for j in (0..@vector.size-1)
            if j+1 != @vector.size
                if @vector[j+1] < @vector[j]
                    @vector[j], @vector[j+1] = @vector[j+1], @vector[j]
                end
            end
        end
    end
    @vector
end
